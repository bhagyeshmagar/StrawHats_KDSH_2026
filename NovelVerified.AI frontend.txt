/**
 * CLAIM DETAIL MODAL
 */
function ClaimDetailModal({ claim, onClose }) {
  const modalContent = useRef(null);

  // Close on click outside
  useEffect(() => {
    const handleClick = (e) => {
      if (modalContent.current && !modalContent.current.contains(e.target)) {
        onClose();
      }
    };
    document.addEventListener('mousedown', handleClick);
    return () => document.removeEventListener('mousedown', handleClick);
  }, [onClose]);

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-slate-900/60 backdrop-blur-sm animate-in fade-in duration-200">
      <div 
        ref={modalContent}
        className="bg-white w-full max-w-3xl max-h-[90vh] rounded-2xl shadow-2xl overflow-hidden flex flex-col animate-in zoom-in-95 duration-200"
      >
        <div className="p-6 border-b border-slate-100 flex items-start justify-between bg-slate-50/50">
          <div>
            <div className="flex items-center gap-3 mb-2">
              <span className={`
                px-3 py-1 rounded-full text-xs font-bold tracking-wide uppercase border
                ${claim.verdict === 'SUPPORTED' ? 'bg-green-100 text-green-700 border-green-200' : 'bg-red-100 text-red-700 border-red-200'}
              `}>
                {claim.verdict}

            <h2 className="text-xl font-bold text-slate-900">"{claim.claim}"</h2>

/**
 * CHAT INTERFACE
 */
function ChatInterface() {
  const [messages, setMessages] = useState([
    { 
      role: 'system', 
      content: "Greetings. I am the NovelVerified Agent. I can verify character backstories against the original text of literary works. Try asking me to verify a claim.",
      type: 'text'
    }
  ]);
  const [inputValue, setInputValue] = useState('');
  const [isProcessing, setIsProcessing] = useState(false);
  const scrollRef = useRef(null);

  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages]);

  const handleSend = async (e) => {
    e.preventDefault();
    if (!inputValue.trim() || isProcessing) return;

    const userText = inputValue;
    setInputValue('');
    
    // Add User Message
    const newMessages = [...messages, { role: 'user', content: userText, type: 'text' }];
    setMessages(newMessages);
    setIsProcessing(true);

    // SIMULATE PIPELINE 
    // 1. Identify intent (simulated)
    await delay(600);
    setMessages(prev => [...prev, { role: 'system', content: `Parsing claim from input...`, type: 'status' }]);
    
    // 2. Identify book (Simple heuristic for demo)
    let detectedBook = "The Count of Monte Cristo"; // Default for demo
    if (userText.toLowerCase().includes("frankenstein")) detectedBook = "Frankenstein";
    if (userText.toLowerCase().includes("darcy") || userText.toLowerCase().includes("bennet")) detectedBook = "Pride and Prejudice";

    await delay(800);
    setMessages(prev => [...prev, { role: 'system', content: `Searching "${detectedBook}" for semantic matches...`, type: 'status' }]);

    // 3. Retrieval simulation
    await delay(1200);
    setMessages(prev => [...prev, { role: 'system', content: `Retrieved 4 relevant passages. Analyzing with Logic Core...`, type: 'status' }]);

    // 4. Generate Verdict (Randomized for variety in demo)
    await delay(1000);
    
    // Create a dynamic result based on input
    const isQuestion = userText.includes('?');
    const verdict = Math.random() > 0.3 ? 'SUPPORTED' : 'CONTRADICTED';
    
    const resultCard = {
      id: `live-${Date.now()}`,
      claim: isQuestion ? userText.replace('?', '.') : userText, // Convert question to statement for "claim"
      book: detectedBook,
      character: "Identified Character",
      verdict: verdict,
      confidence: 0.85 + (Math.random() * 0.14),
      timestamp: new Date().toISOString(),
      evidence: [
        { text: "...evidence found in the text matching the semantic query...", page: 42 },
        { text: "...confirming the narrative sequence of events...", page: 105 }
      ],
      reasoning: `Based on the retrieval analysis of ${detectedBook}, the claim appears to be ${verdict.toLowerCase()}. The text provides direct evidence linking the character's actions to this outcome.`
    };

    setMessages(prev => [
      ...prev.filter(m => m.type !== 'status'), // Remove status messages
      { role: 'system', content: resultCard, type: 'result' }
    ]);
    
    setIsProcessing(false);
  };

  return (
    <div className="flex flex-col h-full bg-white">
      {/* Messages Area */}
      <div 
        ref={scrollRef}
        className="flex-1 overflow-y-auto p-4 space-y-6 bg-slate-50"
      >
        {messages.map((msg, idx) => (
          <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
            <div className={`
              max-w-[85%] md:max-w-[70%] 
              ${msg.role === 'user' ? 'ml-auto' : 'mr-auto'}
            `}>
              {/* Avatar for System */}
              {msg.role === 'system' && msg.type !== 'status' && (
                <div className="flex items-center gap-2 mb-1">
                  <div className="h-6 w-6 rounded-full bg-blue-600 flex items-center justify-center">
                    <BrainCircuit className="h-3 w-3 text-white" />
                  </div>
                  <span className="text-xs font-bold text-slate-500">NovelVerified AI</span>
                </div>
              )}

              {/* Message Content Bubble */}
              {msg.type === 'text' && (
                <div className={`
                  p-4 rounded-2xl text-sm leading-relaxed shadow-sm
                  ${msg.role === 'user' 
                    ? 'bg-blue-600 text-white rounded-tr-none' 
                    : 'bg-white border border-slate-200 text-slate-700 rounded-tl-none'}
                `}>
                  {msg.content}
                </div>
              )}

              {/* Status Updates (Loading style) */}
              {msg.type === 'status' && (
                <div className="flex items-center gap-3 p-3 text-xs text-slate-500 font-mono bg-slate-100 rounded-lg border border-slate-200 animate-pulse">
                   <Loader2 className="h-3 w-3 animate-spin" />
                   {msg.content}
                </div>
              )}

              {/* Rich Result Card */}
              {msg.type === 'result' && (
                <div className="bg-white rounded-xl border border-slate-200 shadow-md overflow-hidden mt-2">
                  <div className={`h-1.5 w-full ${msg.content.verdict === 'SUPPORTED' ? 'bg-green-500' : 'bg-red-500'}`} />
                  <div className="p-4">
                    <div className="flex justify-between items-start mb-2">
                      <span className={`
                        px-2 py-0.5 rounded text-[10px] font-bold tracking-wider border
                        ${msg.content.verdict === 'SUPPORTED' ? 'bg-green-50 text-green-700 border-green-100' : 'bg-red-50 text-red-700 border-red-100'}
                      `}>
                        {msg.content.verdict}
                      </span>
                      <span className="text-xs text-slate-400">{msg.content.book}</span>
                    </div>
                    <p className="font-semibold text-slate-800 text-sm mb-3">"{msg.content.claim}"</p>
                    <div className="text-xs text-slate-600 bg-slate-50 p-3 rounded-lg border border-slate-100 mb-3">
                      {msg.content.reasoning}
                    </div>
                    <button className="w-full py-2 text-xs font-medium text-blue-600 hover:bg-blue-50 rounded transition-colors border border-blue-100 flex items-center justify-center gap-1">
                      <FileText className="h-3 w-3" /> View Full Dossier
                    </button>
                  </div>
                </div>
              )}
            </div>
          </div>
        ))}
        {isProcessing && (
          <div className="flex justify-start">
             {/* Just a spacer for the auto-scroll */}
          </div>
        )}
      </div>

      {/* Input Area */}
      <div className="p-4 bg-white border-t border-slate-200">
        <form onSubmit={handleSend} className="relative max-w-4xl mx-auto">
          <input
            type="text"
            value={inputValue}
            onChange={(e) => setInputValue(e.target.value)}
            disabled={isProcessing}
            placeholder="Type a claim to verify (e.g., 'Edmond Dantes escaped in a body bag')"
            className="w-full pl-4 pr-12 py-3.5 rounded-xl border border-slate-300 focus:border-blue-500 focus:ring-2 focus:ring-blue-100 outline-none text-sm shadow-sm transition-all"
          />
          <button 
            type="submit"
            disabled={!inputValue.trim() || isProcessing}
